<script>

document.addEventListener('DOMContentLoaded', function() {

// ניהול לשוניות

const navTabs = document.querySelectorAll('.nav-tab');

const tabContents = document.querySelectorAll('.tab-content');

navTabs.forEach(tab => {

tab.addEventListener('click', function() {

// הסרת מצב פעיל מכל הלשוניות

navTabs.forEach(t => t.classList.remove('active'));

tabContents.forEach(c => c.classList.remove('active'));

// הוספת מצב פעיל ללשונית שנלחצה

tab.classList.add('active');

// הצגת התוכן המתאים

const tabId = tab.getAttribute('data-tab');

document.getElementById(tabId).classList.add('active');

});

});

// ניהול השמע

let currentAudio = null; // משתנה לשמירת האודיו הנוכחי

let currentIcon = null; // משתנה לשמירת האייקון הנוכחי

const soundIcons = document.querySelectorAll('.sound-icon');

soundIcons.forEach(icon => {

icon.addEventListener('click', function() {

const wordCard = icon.closest('.word-card');

const word = wordCard.getAttribute('data-word');

// אם יש אודיו שמתנגן כרגע, עוצרים אותו

if (currentAudio) {

currentAudio.pause();

currentAudio.currentTime = 0;

if (currentIcon) {

currentIcon.classList.remove('playing');

}

}

// הוספת אפקט לאייקון

icon.classList.add('playing');

currentIcon = icon;

// יצירת אובייקט אודיו חדש

const audio = new Audio(`audio/${word}.mp3`);

currentAudio = audio;

// השמעת האודיו

audio.play();

// הסרת האפקט בסיום ההשמעה

audio.onended = function() {

icon.classList.remove('playing');

currentAudio = null;

currentIcon = null;

};

// במקרה של שגיאה

audio.onerror = function() {

console.error(`Error playing audio for ${word}`);

icon.classList.remove('playing');

currentAudio = null;

currentIcon = null;

// הצגת התראה במקרה של שגיאה (במצב פיתוח)

alert(`שגיאה בטעינת קובץ השמע: ${word}.mp3`);

};

});

});

});

</script>
